#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) { // 1. 배열, 제한수를 입력받는다.
    int sum = 0;
    int j;
    int i = 0;
    while (i < board.size()) { // 2. 배열의 행 개수만큼 반복한다.
        // 2.1. 현재 행의 열 개수만큼 반복한다.
        j = 0;
        while (j < board[i].size()) {
            // 2.1.1. 현재 행렬이 제한수 이하이면 합을 구한다.
            if(i + j <= k){
                sum += board[i][j];
            }
            j++;
        }
        i++;
    }
    return sum; // 3. 합을 출력한다.
} // 4. 끝낸다.
